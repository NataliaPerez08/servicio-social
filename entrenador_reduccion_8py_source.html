13 """
14 Método encargado de realizar la reducción de dimensionalidad y la regresión logística
15 @args:
16 X: ejemplares a usar

30 # Train model

33 # Make predictions
34 y_pred = model.predict(X_test)
35
36 # View accuracy score
37 accuracy = accuracy_score(y_test, y_pred)
38 print("Precisión del modelo:", accuracy)
39 print(precision_score(y_test, y_pred, average='macro', zero_division=0))
40
41 print(classification_report(y_test, y_pred, zero_division=0))
42
43 return model,accuracy
