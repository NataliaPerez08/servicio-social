182 """

187 eje_X = []

192 for i in range(len(ventanas)):
193 hijos = ventanas[i].children()
194 for j in range(len(hijos)):
195 tipo_hijo = type(hijos[j])
196 # Verifica que el hijo sea un matplotlib
197 if tipo_hijo == matplotlib.backends.backend_qtagg.FigureCanvasQTAgg:
198 # Obtiene la figura
199 fig = hijos[j].figure
200 # Obtiene el eje
201 ax = fig.axes[0]
202 # Obtiene los datos
203 line = ax.lines[0]
204 # Obtiene los datos de x
205 x = line.get_xdata()
206 # Obtiene los datos de y
207 y = line.get_ydata()
208 eje_X = x
209 lista_ys.append(y)
210 # Verifica que el hijo sea un label para obtener el nombre del espectro
211 elif tipo_hijo == QtWidgets.QLabel:
212 text = hijos[j].text()

214 espectros.append(text)
